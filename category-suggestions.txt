1. WP-CLI (Command Line)
------------------------
# Create categories first
wp term create category "Linux"
wp term create category "Apple"
wp term create category "Cloud"

# Assign category to a post by ID

# List all posts with IDs
wp post list --post_type=post --fields=ID,post_title


2. SQL Direct Update (fastest for bulk)
---------------------------------------
-- First, get the term_taxonomy_id for each category:
SELECT t.term_id, t.name, tt.term_taxonomy_id 
FROM wp_terms t 
JOIN wp_term_taxonomy tt ON t.term_id = tt.term_id 
WHERE tt.taxonomy = 'category';

-- Then assign posts to categories:
-- Example: Assign all posts with "Linux" or "Ubuntu" in title to Linux category
INSERT IGNORE INTO wp_term_relationships (object_id, term_taxonomy_id)
SELECT ID, <linux_term_taxonomy_id> 
FROM wp_posts 
WHERE post_type = 'post' 
AND post_status = 'publish'
AND (post_title LIKE '%Linux%' OR post_title LIKE '%Ubuntu%' OR post_title LIKE '%Gnome%' OR post_title LIKE '%XFCE%');
